```C++
class Solution {
private:
    unordered_map<char, int> symbolValues = {
        {'I', 1},
        {'V', 5},
        {'X', 10},
        {'L', 50},
        {'C', 100},
        {'D', 500},
        {'M', 1000},
    };

public:
    int romanToInt(string s) {
        int ans = 0;
        int n = s.length();
        for (int i = 0; i < n; ++i) {
            int value = symbolValues[s[i]];
            if (i < n - 1 && value < symbolValues[s[i + 1]]) {
                ans -= value;
            } else {
                ans += value;
            }
        }
        return ans;
    }
};
```

```c++
#include <iostream>
#include <string>

using namespace std;


const pair<int, string> irs[] = {
{1000, "M"},
{900, "CM"},
{500, "D"},
{400, "CD"},
{100, "C"},
{90, "XC"},
{50, "L"},
{40, "XL"},
{10, "X"},
{9, "IX"},
{5, "V"},
{4, "IV"},
{1, "I"},
};


class Solution {
public:

	int romanToInt(string rs)
	{
		int n=0, m=0;
		for (const auto& ir : irs)
		{
			while (m < rs.size() && rs.substr(m, ir.second.size()) == ir.second)
			{
				n += ir.first;
				m += ir.second.size();
			}
		}
		return n;
	}
};

int main()
{
	Solution solution = Solution();
	cout << solution.romanToInt("MCMXCIV");
}
```
