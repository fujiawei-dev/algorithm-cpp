#include <vector>

using namespace std;


class Solution {
public:
	int removeDuplicates(vector<int>& nums) {
		if (nums.empty()) return 0;
		int i = 0;
		for (int j = 0; j < nums.size(); j++)
		{
			if (nums[i] != nums[j]) nums[++i] = nums[j];
		}
		return i + 1;
	}
};

```Java
class Solution {
    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        int fast = 1, slow = 1;
        while (fast < n) {
            if (nums[fast] != nums[fast - 1]) {
                nums[slow] = nums[fast];
                ++slow;
            }
            ++fast;
        }
        return slow;
    }
}
```
